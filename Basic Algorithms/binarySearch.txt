// Binary Search Pseudocode

int bs(int *arr, int l, int r, int x) {
    if(r < l) {
        return -1;
    } else {
        int m = sizeof(arr)/2;
        if(arr[m] == x) {
            return m;
        } else if (arr[m] < x) {
            bs(arr, m+1, r, x);
        } else {
            bs(arr, l, m, x);
        }
    }
}

Time complexity = O(logn)
